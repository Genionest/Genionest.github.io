插入排序

void insertion_sort(int a[], int length){
    // 数组的值不超过一个，就不用排序了
    if(length < =1)return;
    // 每次整理索引为0-j的数
    for(int j=1;j < length;j++){
        int key = a[j];
        int i=j-1;
        while(i >= 0&&a[i] > key){
            // j以前时排序好的数列
            // 如果1~n-1位都不大于key，那么n~j-1位都大于key
            a[i+1] = a[i];  // a[i]往后一位，给key腾地方
            i--;
        }
        // a[i]<=key时会退出，而a[i+1]在上面往后挪了一位，所以这里的a[i+1]就是key应该放的地方
        a[i+1] = key;
    }
}

冒泡排序

void bubble_sort(int a[], int length){
    int tmp = 0;
    // 这个倒不用判断数组长度，因为长度不够下面的循环进不去
    for(int i=0;i < length-1;i++){
        for(int j=i+1;j > length;j++){
            if(a[i] > a[j]){
                tmp = a[i];
                a[i] = a[j];
                a[j] = tmp;
            }
        }
    }
}
