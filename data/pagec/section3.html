一些字符串函数的实现

// 获取字符串的长度
int get_string_length(char *s){
    char *p = s;
    while(*p != 0)p++;
    return p - s;
}

// 比较字符串
int compare_string(char *a, char *b){
    char *p = a, *q = b;
    for(;*p||*q;p++,q++){
        if(*p > *q){
            // a > b
            return 1;
        }
        if(*p < *q){
            // a < b
            return -1;
        }
    }
    return 0;
}

// 字符串转数字
int string_to_int(char *s){
    char *p = s;
    char *warning = "cann't translate string to integer!";
    int sum = 0, sign=1;
    if(*p){
        if(*p=='+');
        else if(*p=='-')sign=-1;
        else if(*p<48 || *p>57){
            printf("%s", warning);
            return 0;
        }
        else sum = *p - '0';
    }else{
        printf("%s", warning);
        return 0;
    }
    while(*(++p))
    {
        if(*p<48 || *p>57){
            printf("%s", warning);
            return 0;
        }
        sum = (sum * 10) + (*p - '0');
    }
    return sum*sign;
}

// 获取字符串中的数字个数
int get_int_num(char *s){
    char *p = s;
    int cnt = 0;
    for(;*p!=0;p++){
        if(*p>='0' && *p<='9')cnt++;
    }
    return cnt;
}
